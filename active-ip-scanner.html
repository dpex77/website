import subprocess import platform from ipaddress import ip_network from threading import Thread # User input subnet = input("Enter subnet to scan (e.g., 192.168.209.0/24): ") # Convert subnet string to network network = ip_network(subnet, strict=False) # Determine ping parameters based on OS param = "-n" if platform.system().lower() == "windows" else "-c" timeout_param = "-w" if platform.system().lower() == "windows" else "-W" print(f"Scanning subnet {network}...\n") # Function to ping a single IP def ping(ip): result = subprocess.run( ["ping", param, "1", timeout_param, "1", str(ip)], stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL ) if result.returncode == 0: print(f"{ip} is ACTIVE") # Create and start threads threads = [] for ip in network.hosts(): t = Thread(target=ping, args=(ip,)) threads.append(t) t.start() # Wait for all threads to finish for t in threads: t.join() print("\nScan complete.")
